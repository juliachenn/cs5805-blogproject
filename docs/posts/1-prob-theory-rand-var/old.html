def clean_data(data):
    lem = WordNetLemmatizer()
    data = data.lower()
    cleaned_data = ''

    

    for word in data.split(" "):
        word = lem.lemmatize(word)
        cleaned_data = cleaned_data + word + ' '
    return cleaned_data
